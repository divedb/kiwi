31// Thin wrapper around an OS-level file.
32// Note that this class does not provide any support for asynchronous IO, other
33// than the ability to create asynchronous handles on Windows.
34//
35// Note about const: this class does not attempt to determine if the underlying
36// file system object is affected by a particular method in order to consider
37// that method const or not. Only methods that deal with member variables in an
38// obvious non-modifying way are marked as const. Any method that forward calls
39// to the OS is not considered const, even if there is no apparent change to
40// member variables.
41//
42// On POSIX, if the given file is a symbolic link, most of the methods apply to
43// the file that the symbolic link resolves to.
44class BASE_EXPORT File {
45 public:
46 // FLAG_(OPEN|CREATE).* are mutually exclusive. You should specify exactly one
47 // of the five (possibly combining with other flags) when opening or creating
48 // a file.
49 // FLAG_(WRITE|APPEND) are mutually exclusive. This is so that APPEND behavior
50 // will be consistent with O_APPEND on POSIX.
51 enum Flags : uint32_t {
52 FLAG_OPEN = 1 << 0, // Opens a file, only if it exists.
53 FLAG_CREATE = 1 << 1, // Creates a new file, only if it does not
54 // already exist.
55 FLAG_OPEN_ALWAYS = 1 << 2, // May create a new file.
56 FLAG_CREATE_ALWAYS = 1 << 3, // May overwrite an old file.
57 FLAG_OPEN_TRUNCATED = 1 << 4, // Opens a file and truncates it, only if it
58 // exists.
59 FLAG_READ = 1 << 5,
60 FLAG_WRITE = 1 << 6,
61 FLAG_APPEND = 1 << 7,
62 FLAG_WIN_EXCLUSIVE_READ = 1 << 8, // Windows only. Opposite of SHARE.
63 FLAG_WIN_EXCLUSIVE_WRITE = 1 << 9, // Windows only. Opposite of SHARE.
64 FLAG_ASYNC = 1 << 10,
65 FLAG_WIN_TEMPORARY = 1 << 11, // Windows only.
66 FLAG_WIN_HIDDEN = 1 << 12, // Windows only.
67 FLAG_DELETE_ON_CLOSE = 1 << 13,
68 FLAG_WRITE_ATTRIBUTES = 1 << 14, // File opened in a mode allowing writing
69 // attributes, such as with SetTimes().
70 FLAG_WIN_SHARE_DELETE = 1 << 15, // Windows only.
71 FLAG_TERMINAL_DEVICE = 1 << 16, // Serial port flags.
72 FLAG_WIN_BACKUP_SEMANTICS = 1 << 17, // Windows only.
73 FLAG_WIN_EXECUTE = 1 << 18, // Windows only.
74 FLAG_WIN_SEQUENTIAL_SCAN = 1 << 19, // Windows only.
75 FLAG_CAN_DELETE_ON_CLOSE = 1 << 20, // Requests permission to delete a file
76 // via DeleteOnClose() (Windows only).
77 // See DeleteOnClose() for details.
78 FLAG_WIN_NO_EXECUTE =
79 1 << 21, // Windows only. Marks the file with a deny ACE that prevents
80 // opening the file with EXECUTE access. Cannot be used with
81 // FILE_WIN_EXECUTE flag. See also PreventExecuteMapping.
82 };

301 // Attempts to take an exclusive write lock on the file. Returns immediately
302 // (i.e. does not wait for another process to unlock the file). If the lock
303 // was obtained, the result will be FILE_OK. A lock only guarantees
304 // that other processes may not also take a lock on the same file with the
305 // same API - it may still be opened, renamed, unlinked, etc.
306 //
307 // Common semantics:
308 // * Locks are held by processes, but not inherited by child processes.
309 // * Locks are released by the OS on file close or process termination.
310 // * Locks are reliable only on local filesystems.
311 // * Duplicated file handles may also write to locked files.
312 // Windows-specific semantics:
313 // * Locks are mandatory for read/write APIs, advisory for mapping APIs.
314 // * Within a process, locking the same file (by the same or new handle)
315 // will fail.
316 // POSIX-specific semantics:
317 // * Locks are advisory only.
318 // * Within a process, locking the same file (by the same or new handle)
319 // will succeed. The new lock replaces the old lock.
320 // * Closing any descriptor on a given file releases the lock.
321 Error Lock(LockMode mode);
322
323 // Unlock a file previously locked.
324 Error Unlock();
325
326#endif // !BUILDFLAG(IS_FUCHSIA)

347 // Sets or clears the DeleteFile disposition on the file. Returns true if
348 // the disposition was set or cleared, as indicated by |delete_on_close|.
349 //
350 // Microsoft Windows deletes a file only when the DeleteFile disposition is
351 // set on a file when the last handle to the last underlying kernel File
352 // object is closed. This disposition is be set by:
353 // - Calling the Win32 DeleteFile function with the path to a file.
354 // - Opening/creating a file with FLAG_DELETE_ON_CLOSE and then closing all
355 // handles to that File object.
356 // - Opening/creating a file with FLAG_CAN_DELETE_ON_CLOSE and subsequently
357 // calling DeleteOnClose(true).
358 //
359 // In all cases, all pre-existing handles to the file must have been opened
360 // with FLAG_WIN_SHARE_DELETE. Once the disposition has been set by any of the
361 // above means, no new File objects can be created for the file.
362 //
363 // So:
364 // - Use FLAG_WIN_SHARE_DELETE when creating/opening a file to allow another
365 // entity on the system to cause it to be deleted when it is closed. (Note:
366 // another entity can delete the file the moment after it is closed, so not
367 // using this permission doesn't provide any protections.)
368 // - Use FLAG_DELETE_ON_CLOSE for any file that is to be deleted after use.
369 // The OS will ensure it is deleted even in the face of process termination.
370 // Note that it's possible for deletion to be cancelled via another File
371 // object referencing the same file using DeleteOnClose(false) to clear the
372 // DeleteFile disposition after the original File is closed.
373 // - Use FLAG_CAN_DELETE_ON_CLOSE in conjunction with DeleteOnClose() to alter
374 // the DeleteFile disposition on an open handle. This fine-grained control
375 // allows for marking a file for deletion during processing so that it is
376 // deleted in the event of untimely process termination, and then clearing
377 // this state once the file is suitable for persistence.
378 bool DeleteOnClose(bool delete_on_close);

405 // This function can be used to augment `flags` with the correct flags
406 // required to create a File that can be safely passed to an untrusted
407 // process. It must be called if the File is intended to be transferred to an
408 // untrusted process, but can still be safely called even if the File is not
409 // intended to be transferred.
410 static constexpr uint32_t AddFlagsForPassingToUntrustedProcess(
411 uint32_t flags) {
412 if (flags & File::FLAG_WRITE || flags & File::FLAG_APPEND ||
413 flags & File::FLAG_WRITE_ATTRIBUTES) {
414 flags |= File::FLAG_WIN_NO_EXECUTE;
415 }
416 return flags;
417 }
