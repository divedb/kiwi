23// Annotates a pointer and size directing MSAN to treat that memory region as
24// fully initialized. Useful for e.g. code that deliberately reads uninitialized
25// data, such as a GC scavenging root set pointers from the stack.
26//
27// See also:
28// https://github.com/google/sanitizers/wiki/MemorySanitizer
29//
30// Usage:
31// ```
32// T* ptr = ...;
33// // After the next statement, MSAN will assume `ptr` points to an
34// // initialized `T`.
35// MSAN_UNPOISON(ptr, sizeof(T));
36// ```
37#if defined(MEMORY_SANITIZER) && !BUILDFLAG(IS_NACL)
38#include <sanitizer/msan_interface.h>
39#define MSAN_UNPOISON(p, size) __msan_unpoison(p, size)
40#else
41#define MSAN_UNPOISON(p, size)
42#endif

44// Annotates a pointer and size directing MSAN to check whether that memory
45// region is initialized, as if it was being read from. If any bits are
46// uninitialized, crashes with an MSAN report. Useful for e.g. sanitizing data
47// MSAN won't be able to track, such as data that is about to be passed to
48// another process via shared memory.
49//
50// See also:
51// https://www.chromium.org/developers/testing/memorysanitizer/#debugging-msan-reports
52//
53// Usage:
54// ```
55// T* ptr = ...;
56// // The following line will crash at runtime in MSAN builds if `ptr` does
57// // not point to an initialized `T`.
58// MSAN_CHECK_MEM_IS_INITIALIZED(ptr, sizeof(T));
59// ```
60#if defined(MEMORY_SANITIZER) && !BUILDFLAG(IS_NACL)
61#define MSAN_CHECK_MEM_IS_INITIALIZED(p, size) \
62 __msan_check_mem_is_initialized(p, size)
63#else
64#define MSAN_CHECK_MEM_IS_INITIALIZED(p, size)
65#endif

152// Annotates a codepath suppressing static analysis along that path. Useful when
153// code is safe in practice for reasons the analyzer can't detect, e.g. because
154// the condition leading to that path guarantees a param is non-null.
155//
156// Usage:
157// ```
158// if (cond) {
159// ANALYZER_SKIP_THIS_PATH();
160// // Static analysis will be disabled for the remainder of this block.
161// delete ptr;
162// }
163// ```
164#if defined(__clang_analyzer__)
165inline constexpr bool AnalyzerNoReturn()
166#if KIWI_HAS_ATTRIBUTE(analyzer_noreturn)
167 __attribute__((analyzer_noreturn))
168#endif
169{
170 return false;
171}
172#define ANALYZER_SKIP_THIS_PATH() static_cast<void>(::AnalyzerNoReturn())
173#else
174// The above definition would be safe even outside the analyzer, but defining
175// the macro away entirely avoids the need for the optimizer to eliminate it.
176#define ANALYZER_SKIP_THIS_PATH()
177#endif

179// Annotates a condition directing static analysis to assume it is always true.
180// Evaluates to the provided `arg` as a `bool`.
181//
182// Usage:
183// ```
184// // Static analysis will assume the following condition always holds.
185// if (ANALYZER_ASSUME_TRUE(cond)) ...
186// ```
187#if defined(__clang_analyzer__)
188inline constexpr bool AnalyzerAssumeTrue(bool arg) {
189 return arg || AnalyzerNoReturn();
190}
191#define ANALYZER_ASSUME_TRUE(arg) ::AnalyzerAssumeTrue(!!(arg))
192#else
193// Again, the above definition is safe, this is just simpler for the optimizer.
194#define ANALYZER_ASSUME_TRUE(arg) (arg)
195#endif

197// Determines whether a type is trivially relocatable, i.e. a move-and-destroy
198// sequence can safely be replaced with `memcpy()`. This is true of types with
199// trivial copy or move construction plus trivial destruction, as well as types
200// marked `TRIVIAL_ABI`. Useful to optimize container implementations.
201//
202// See also:
203// https://www.open-std.org/jtc1/sc22/wg21/docs/papers/2023/p1144r8.html
204// https://clang.llvm.org/docs/LanguageExtensions.html#:~:text=__is_trivially_relocatable
205//
206// Usage:
207// ```
208// if constexpr (IS_TRIVIALLY_RELOCATABLE(T)) {
209// // This block will only be executed if type `T` is trivially relocatable.
210// }
211// ```
212#if KIWI_HAS_BUILTIN(__is_trivially_relocatable)
213#define IS_TRIVIALLY_RELOCATABLE(t) __is_trivially_relocatable(t)
214#else
215#define IS_TRIVIALLY_RELOCATABLE(t) false
216#endif

218// Annotates code indicating that it should be permanently exempted from
219// `-Wunsafe-buffer-usage`. For temporary cases such as migrating callers to
220// safer patterns, use `UNSAFE_TODO()` instead; see documentation there.
221//
222// All calls to functions annotated with `UNSAFE_BUFFER_USAGE` must be marked
223// with one of these two macros; they can also be used around pointer
224// arithmetic, pointer subscripting, and the like.
225//
226// ** USE OF THIS MACRO SHOULD BE VERY RARE.** Using this macro indicates that
227// the compiler cannot verify that the code avoids OOB, and manual review is
228// required. Even with manual review, it's easy for assumptions to change and
229// security bugs to creep in over time. Prefer safer patterns instead.
230//
231// Usage should wrap the minimum necessary code, and *must* include a
232// `// SAFETY: ...` comment that explains how the code guarantees safety or
233// meets the requirements of called `UNSAFE_BUFFER_USAGE` functions. Guarantees
234// must be manually verifiable by the Chrome security team using only local
235// invariants; contact security@chromium.org to schedule such a review. Valid
236// invariants include:
237// - Runtime conditions or `CHECK()`s nearby
238// - Invariants guaranteed by types in the surrounding code
239// - Invariants guaranteed by function calls in the surrounding code
240// - Caller requirements, if the containing function is itself annotated with
241// `UNSAFE_BUFFER_USAGE`; this is less safe and should be a last resort
242//
243// See also:
244// https://chromium.googlesource.com/chromium/src/+/main/docs/unsafe_buffers.md
245// https://clang.llvm.org/docs/SafeBuffers.html
246// https://clang.llvm.org/docs/DiagnosticsReference.html#wunsafe-buffer-usage
247//
248// Usage:
249// ```
250// // The following call will not trigger a compiler warning even if `Func()`
251// // is annotated `UNSAFE_BUFFER_USAGE`.
252// return UNSAFE_BUFFERS(Func(input, end));
253// ```
254//
255// Test for `__clang__` directly, as there's no `__has_pragma` or similar (see
256// https://github.com/llvm/llvm-project/issues/51887).
257#if defined(__clang__)
258// Disabling `clang-format` allows each `_Pragma` to be on its own line, as
259// recommended by https://gcc.gnu.org/onlinedocs/cpp/Pragmas.html.
260// clang-format off
261#define UNSAFE_BUFFERS(...) \
262 _Pragma("clang unsafe_buffer_usage begin") \
263 __VA_ARGS__ \
264 _Pragma("clang unsafe_buffer_usage end")
265// clang-format on
266#else
267#define UNSAFE_BUFFERS(...) __VA_ARGS__
268#endif

302#ifdef __clang__

309 KIWI_GNU_DISABLE_WARNING(warning_name)
310#endif

312#elif defined(_MSC_VER)
313#define KIWI_PUSH_WARNING __pragma(warning(push))
