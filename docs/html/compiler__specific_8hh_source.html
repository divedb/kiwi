413// Annotates a variable indicating that its storage should not be filled with a
414// fixed pattern when uninitialized.
415//
416// The `init_stack_vars` gn arg (enabled on most build configs) causes the
417// compiler to generate code that writes a fixed pattern into uninitialized
418// parts of all local variables, to mitigate security risks. In most cases, e.g.
419// when such memory is either never accessed or will be initialized later before
420// reading, the compiler is able to remove the additional stores, and any
421// remaining stores are unlikely to affect program performance.
422//
423// If hot code suffers unavoidable perf penalties, this can disable the
424// pattern-filling there. This should only be done when necessary, since reads
425// from uninitialized variables are not only UB, they can in practice allow
426// attackers to control logic by pre-filling the variable's memory with a
427// desirable value.
428//
429// NOTE: This behavior also increases the likelihood the compiler will generate
430// `memcpy()`/`memset()` calls to init variables. If this causes link errors for
431// targets that don't link against the CRT, this macro can help; you may instead
432// want 'configs -= [ "//build/config/compiler:default_init_stack_vars" ]' in
433// the relevant .gn file to disable this on the whole target.
434//
435// See also:
436// https://source.chromium.org/chromium/chromium/src/+/main:build/config/compiler/BUILD.gn;l=3088;drc=24ccaf63ff5b1883be1ebe5f979d917ce28b0131
437// https://clang.llvm.org/docs/ClangCommandLineReference.html#cmdoption-clang-ftrivial-auto-var-init
438// https://clang.llvm.org/docs/AttributeReference.html#uninitialized
439//
440// Usage:
441// ```
442// // The following line declares `i` without ensuring it initially contains
443// // any particular pattern.
444// STACK_UNINITIALIZED int i;
445// ```
446#if __has_cpp_attribute(clang::uninitialized)
447#define STACK_UNINITIALIZED [[clang::uninitialized]]
448#elif __has_cpp_attribute(gnu::uninitialized)
449#define STACK_UNINITIALIZED [[gnu::uninitialized]]
450#else
451#define STACK_UNINITIALIZED
452#endif

549// Annotates a type as being suitable for passing in registers despite having a
550// non-trivial copy or move constructor or destructor. This requires the type
551// not be concerned about its address remaining constant, be safely usable after
552// copying its memory, and have a destructor that may be safely omitted on
553// moved-from instances; an example is `std::unique_ptr`. Unnecessary if the
554// copy/move constructor(s) and destructor are unconditionally trivial; likely
555// ineffective if the type is too large to be passed in one or two registers
556// with the target ABI. However, annotating a type this way will also cause
557// `IS_TRIVIALLY_RELOCATABLE()` to return true for that type, and so may be
558// desirable even for large types, if they are placed in containers that
559// optimize based on that check.
560//
561// NOTE: Use with caution; this has subtle effects on constructor/destructor
562// ordering. When used with types passed or returned by value, values may be
563// constructed in the source stack frame, passed in a register, and then used
564// and destroyed in the target stack frame.

566// See also:
567// https://clang.llvm.org/docs/AttributeReference.html#trivial-abi
568// https://libcxx.llvm.org/docs/DesignDocs/UniquePtrTrivialAbi.html
569//
570// Usage:
571// ```
572// // Instances of type `S` will be eligible to be passed in registers despite
573// // `S`'s nontrivial destructor.
574// struct TRIVIAL_ABI S { ~S(); }
575// ```
576#if __has_cpp_attribute(clang::trivial_abi)
577#define TRIVIAL_ABI [[clang::trivial_abi]]
578#else
579#define TRIVIAL_ABI

773// Annotates a function or variable to indicate that it should have weak
774// linkage. Useful for library code that wants code linking against it to be
775// able to override its functionality; inside a single target, this is better
776// accomplished via virtual methods and other more standard mechanisms.
777//
778// Any weak definition of a symbol will be overridden at link time by a non-weak
779// definition. Marking a `const` or `constexpr` variable weak makes it no longer
780// be considered a compile-time constant, since its value may be different after
781// linking.
782//
783// Multiple weak definitions of a symbol may exist, in which case the linker is
784// free to select any when there are no non-weak definitions. Like with symbols
785// marked `inline`, this can lead to subtle, difficult-to-diagnose bugs if not
786// all definitions are identical.
787//
788// A weak declaration that has no definitions at link time will be linked as if
789// the corresponding address is null. Therefore library code can use weak
790// declarations and conditionals to allow consumers to provide optional
791// customizations.
792//
793// See also:
794// https://clang.llvm.org/docs/AttributeReference.html#weak
795//
796// Usage:
797// ```
798// // The following definition defaults `x` to 10, but allows other object
799// // files to override its value. Thus, despite `constexpr`, `x` is not
800// // considered a compile-time constant (and cannot be used in a `constexpr`
801// // context).
802// extern const int x;
803// WEAK_SYMBOL constexpr int x = 10;
804//
805// // The following declaration allows linking to occur whether a definition
806// // of `Func()` is provided or not; if none is present, `&Func` will
807// // evaluate to `nullptr` at runtime.
808// WEAK_SYMBOL void Func();
809//
810// // The following definition provides a default implementation of `Func2()`,
811// // but allows other object files to override.
812// WEAK_SYMBOL void Func2() { ... }
813// ```
814#if __has_cpp_attribute(gnu::weak)
815#define WEAK_SYMBOL [[gnu::weak]]
816#else
817#define WEAK_SYMBOL
818#endif

1061// Annotates a function restricting its availability based on compile-time
1062// information in the evaluated context. Useful to convert runtime errors to
1063// compile-time errors if functions' arguments are always known at compile time.
1064//
1065// SFINAE and `requires` clauses can restrict function availability based on the
1066// unevaluated context (type information and syntactic correctness). This
1067// provides a similar capability based on the evaluated context (variable
1068// values). If the condition fails, or cannot be determined at compile time, the
1069// function is excluded from the overload set.
1070//
1071// Some use cases could be satisfied without this by marking the function
1072// `consteval` and breaking compile when the condition fails (e.g. via
1073// `CHECK()`/`assert()`). However, `ENABLE_IF_ATTR()` is generally superior:
1074// - Not all desired functions can be made `consteval`; e.g. most
1075// constructors.
1076// - The error message in the macro case is clearer and more actionable.
1077// - `ENABLE_IF_ATTR()` interacts better with template metaprogramming.
1078//
1079// See also:
1080// https://clang.llvm.org/docs/AttributeReference.html#enable-if
1081// https://github.com/chromium/subspace/issues/266
1082//
1083// Usage:
1084// ```
1085// void NotConsteval(int a) {
1086// assert(a > 0);
1087// }
1088// consteval void WithoutEnableIf(int a) {
1089// assert(a > 0);
1090// }
1091// void WithEnableIf(int a) ENABLE_IF_ATTR(a > 0, "arg must be positive") {}
1092// void Func(int i) {
1093// // Compiles; assertion fails at runtime.
1094// NotConsteval(-1);
1095//
1096// // Will not compile; diagnosed as not a constant expression.
1097// WithoutEnableIf(-1);
1098//
1099// // Will not compile; diagnosed as no matching function call with
1100// // "note: candidate disabled: arg must be positive".
1101// WithEnableIf(-1);
1102//
1103// // Will not compile (same reason). Marking `Func()` as
1104// // `ENABLE_IF_ATTR(i > 0, ...)` will not help; the compiler's analysis is
1105// // not sufficiently sophisticated to propagate this constraint.
1106// WithEnableIf(i);
1107// }
1108// ```
1109#if HAS_ATTRIBUTE(enable_if)
1110#define ENABLE_IF_ATTR(cond, msg) __attribute__((enable_if(cond, msg)))
1111#else
1112#define ENABLE_IF_ATTR(cond, msg)
1113#endif
