60// always_false
61//
62// A variable template that is always false but requires template arguments to
63// be provided (which are then ignored). This is useful in very specific cases
64// where we want type-dependent expressions to defer static_assert's.
65//
66// A common use-case is for exhaustive constexpr if branches:
67//
68// template <typename T>
69// void foo(T value) {
70// if constexpr (std::is_integral_v<T>) foo_integral(value);
71// else if constexpr (std::is_same_v<T, std::string>) foo_string(value);
72// else static_assert(always_false<T>, "Unsupported type");
73// }
74//
75// If we had used static_assert(false), then this would always fail to compile,
76// even if foo is never instantiated!
77//
78// Another use case is if a template that is expected to always be specialized
79// is erroneously instantiated with the base template.
80//
81// template <typename T>
82// struct Foo {
83// static_assert(always_false<T>, "Unsupported type");
84// };
85// template <>
86// struct Foo<int> {};
87//
88// Foo<int> a; // fine
89// Foo<std::string> b; // fails! And you get a nice (custom) error message
90//
91// This is similar to leaving the base template undefined but we get a nicer
92// compiler error message with static_assert.
93template <typename...>
94inline constexpr bool always_false = false;

189struct is_constexpr_default_constructible_ {
190 template <typename T>
191 static constexpr auto make(tag_t<T>) -> decltype(void(T()), 0) {
192 return (void(T()), 0);
193 }
194 // second param should just be: int = (void(T()), 0)
195 // but under clang 10, crash: https://bugs.llvm.org/show_bug.cgi?id=47620
196 // and, with assertions disabled, expectation failures showing compiler
197 // deviation from the language spec
198 // xcode renumbers clang versions so detection is tricky, but, if detection
199 // were desired, a combination of __apple_build_version__ and __clang_major__
200 // may be used to reduce frontend overhead under correct compilers: clang 12
201 // under xcode and clang 10 otherwise
202 template <typename T, int = make(tag<T>)>
203 static std::true_type sfinae(T*);
204 static std::false_type sfinae(void*);
205 template <typename T>
206 static constexpr bool apply =
207 !require_sizeof<T> || decltype(sfinae(static_cast<T*>(nullptr)))::value;
208};
