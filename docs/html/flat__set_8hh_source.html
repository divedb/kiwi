15// flat_set is a container with a std::set-like interface that stores its
16// contents in a sorted container, by default a vector.
17//
18// Its implementation mostly tracks the corresponding standardization proposal
19// https://wg21.link/P1222.
20//
21// Please see //base/containers/README.md for an overview of which container
22// to select.
23//
24// PROS
25//
26// - Good memory locality.
27// - Low overhead, especially for smaller sets.
28// - Performance is good for more workloads than you might expect (see
29// overview link above).
30// - Supports C++14 set interface.
31//
32// CONS
33//
34// - Inserts and removals are O(n).
35//
36// IMPORTANT NOTES
37//
38// - Iterators are invalidated across mutations.
39// - If possible, construct a flat_set in one operation by inserting into
40// a container and moving that container into the flat_set constructor.
41// - For multiple removals use kiwi::EraseIf() which is O(n) rather than
42// O(n * removed_items).
43//
44// QUICK REFERENCE
45//
46// Most of the core functionality is inherited from flat_tree. Please see
47// flat_tree.h for more details for most of these functions. As a quick
48// reference, the functions available are:
49//
50// Constructors (inputs need not be sorted):
51// flat_set(const flat_set&);
52// flat_set(flat_set&&);
53// flat_set(InputIterator first, InputIterator last,
54// const Compare& compare = Compare());
55// flat_set(const container_type& items,
56// const Compare& compare = Compare());
57// flat_set(container_type&& items,
58// const Compare& compare = Compare()); // Re-use storage.
59// flat_set(std::initializer_list<value_type> ilist,
60// const Compare& comp = Compare());
61//
62// Constructors (inputs need to be sorted):
63// flat_set(sorted_unique_t,
64// InputIterator first, InputIterator last,
65// const Compare& compare = Compare());
66// flat_set(sorted_unique_t,
67// const container_type& items,
68// const Compare& compare = Compare());
69// flat_set(sorted_unique_t,
70// container_type&& items,
71// const Compare& compare = Compare()); // Re-use storage.
72// flat_set(sorted_unique_t,
73// std::initializer_list<value_type> ilist,
74// const Compare& comp = Compare());

77// flat_set& operator=(const flat_set&);

156// Utility function to simplify constructing a flat_set from a fixed list
157// of keys. The keys are obtained by applying the projection |proj| to the
158// |unprojected_elements|. The set's keys are sorted by |comp|.
159//
160// Example usage (creates a set {16, 9, 4, 1}):
161// auto set = kiwi::MakeFlatSet<int>(
162// std::vector<int>{1, 2, 3, 4}, [](int i, int j) { return i > j; },
163// [](int i) { return i * i; });
164template <class Key, class Compare = std::less<>,
165 class Container = std::vector<Key>, class InputContainer,
166 class Projection = std::identity>
167constexpr flat_set<Key, Compare, Container> MakeFlatSet(
168 const InputContainer& unprojected_elements, const Compare& comp = Compare(),
169 const Projection& proj = Projection()) {
170 Container elements;
171 internal::ReserveIfSupported(elements, unprojected_elements);
172 std::ranges::transform(unprojected_elements, std::back_inserter(elements),
173 proj);
174 return flat_set<Key, Compare, Container>(std::move(elements), comp);
175}
