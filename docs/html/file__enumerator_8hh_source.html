31// A class for enumerating the files in a provided path. The order of the
32// results is not guaranteed.
33//
34// This is blocking. Do not use on critical threads.
35//
36// Example:
37//
38// kiwi::FileEnumerator e(my_dir, false, kiwi::FileEnumerator::FILES,
39// FILE_PATH_LITERAL("*.txt"));
40// Using `ForEach` with a lambda:
41// e.ForEach([](const kiwi::FilePath& item) {...});
42// Using a `for` loop:
43// for (kiwi::FilePath name = e.Next(); !name.empty(); name = e.Next())
44// ...
45class BASE_EXPORT FileEnumerator {
46 public:
47 // Note: copy & assign supported.
48 class BASE_EXPORT FileInfo {
49 public:
50 FileInfo();
51#if BUILDFLAG(IS_ANDROID)
52 // Android has both posix paths, and Content-URIs. It will use the linux /
53 // posix code for posix paths where a FileInfo() object is constructed and
54 // then `stat_` is populated via fstat() and used for IsDirectory(),
55 // GetSize(), GetLastModifiedTime(). Content-URIs provide all values in this
56 // constructor and writes `is_directory`, `size` and `time` to `stat_`.
57 FileInfo(kiwi::FilePath content_uri, kiwi::FilePath filename,
58 bool is_directory, off_t size, Time time);
59#endif
60 FileInfo(const FileInfo& that);
61 FileInfo& operator=(const FileInfo& that);
62 FileInfo(FileInfo&& that);
63 FileInfo& operator=(FileInfo&& that);
64 ~FileInfo();
65
66 bool IsDirectory() const;
67
68 // The name of the file. This will not include any path information. This
69 // is in constrast to the value returned by FileEnumerator.Next() which
70 // includes the |root_path| passed into the FileEnumerator constructor.
71 FilePath GetName() const;
72
73#if BUILDFLAG(IS_ANDROID)
74 // Display names of subdirs.
75 const std::vector<std::string>& subdirs() const { return subdirs_; }
76#endif
77
78 int64_t GetSize() const;
79
80 // On POSIX systems, this is rounded down to the second.
81 Time GetLastModifiedTime() const;
82
83#if BUILDFLAG(IS_WIN)
84 // Note that the cAlternateFileName (used to hold the "short" 8.3 name)
85 // of the WIN32_FIND_DATA will be empty. Since we don't use short file
86 // names, we tell Windows to omit it which speeds up the query slightly.
87 const WIN32_FIND_DATA& find_data() const {
88 return *ChromeToWindowsType(&find_data_);

90#elif BUILDFLAG(IS_POSIX) || BUILDFLAG(IS_FUCHSIA)
91 const stat_wrapper_t& stat() const { return stat_; }
92#endif
93

97#if BUILDFLAG(IS_ANDROID)

100#endif

103#elif BUILDFLAG(IS_POSIX) || BUILDFLAG(IS_FUCHSIA)
104 stat_wrapper_t stat_;
105 FilePath filename_;
106#endif
107 };

109 enum FileType {
110 FILES = 1 << 0,
111 DIRECTORIES = 1 << 1,
112 INCLUDE_DOT_DOT = 1 << 2,
113
114 // Report only the names of entries and not their type, size, or
115 // last-modified time. May only be used for non-recursive enumerations, and
116 // implicitly includes both files and directories (neither of which may be
117 // specified). When used, an enumerator's `GetInfo()` method must not be
118 // called.
119 NAMES_ONLY = 1 << 3,
120
121#if BUILDFLAG(IS_POSIX) || BUILDFLAG(IS_FUCHSIA)
122 SHOW_SYM_LINKS = 1 << 4,
123#endif
124 };

126 // Search policy for intermediate folders.
127 enum class FolderSearchPolicy {
128 // Recursive search will pass through folders whose names match the
129 // pattern. Inside each one, all files will be returned. Folders with names
130 // that do not match the pattern will be ignored within their interior.
131 MATCH_ONLY,
132 // Recursive search will pass through every folder and perform pattern
133 // matching inside each one.
134 ALL,
135 };
136
137 // Determines how a FileEnumerator handles errors encountered during
138 // enumeration. When no ErrorPolicy is explicitly set, FileEnumerator defaults
139 // to IGNORE_ERRORS.
140 enum class ErrorPolicy {
141 // Errors are ignored if possible and FileEnumerator returns as many files
142 // as it is able to enumerate.
143 IGNORE_ERRORS,
144
145 // Any error encountered during enumeration will terminate the enumeration
146 // immediately. An error code indicating the nature of a failure can be
147 // retrieved from |GetError()|.
148 STOP_ENUMERATION,
149 };
150
151 // |root_path| is the starting directory to search for. It may or may not end
152 // in a slash.
153 //
154 // If |recursive| is true, this will enumerate all matches in any
155 // subdirectories matched as well. It does a breadth-first search, so all
156 // files in one directory will be returned before any files in a
157 // subdirectory.
158 //
159 // |file_type|, a bit mask of FileType, specifies whether the enumerator
160 // should match files, directories, or both.
161 //
162 // |pattern| is an optional pattern for which files to match. This
163 // works like shell globbing. For example, "*.txt" or "Foo???.doc".
164 // However, be careful in specifying patterns that aren't cross platform
165 // since the underlying code uses OS-specific matching routines. In general,
166 // Windows matching is less featureful than others, so test there first.
167 // If unspecified, this will match all files.
168 //
169 // |folder_search_policy| optionally specifies a search behavior. Refer to
170 // |FolderSearchPolicy| for a list of folder search policies and the meaning
171 // of them. If |recursive| is false, this parameter has no effect.
172 //
173 // |error_policy| optionally specifies the behavior when an error occurs.
174 // Refer to |ErrorPolicy| for a list of error policies and the meaning of
175 // them.
176 FileEnumerator(const FilePath& root_path, bool recursive, int file_type);
177 FileEnumerator(const FilePath& root_path, bool recursive, int file_type,
178 const FilePath::StringType& pattern);
179 FileEnumerator(const FilePath& root_path, bool recursive, int file_type,
180 const FilePath::StringType& pattern,
181 FolderSearchPolicy folder_search_policy);
182 FileEnumerator(const FilePath& root_path, bool recursive, int file_type,
183 const FilePath::StringType& pattern,
184 FolderSearchPolicy folder_search_policy,
185 ErrorPolicy error_policy);
186 FileEnumerator(const FileEnumerator&) = delete;
187 FileEnumerator& operator=(const FileEnumerator&) = delete;
188 ~FileEnumerator();
189
190 // Calls `ref` synchronously for each path found by the `FileEnumerator`. Each
191 // path will incorporate the `root_path` passed in the constructor:
192 // "<root_path>/file_name.txt". If the `root_path` is absolute, then so will
193 // be the paths provided in the `ref` invocations.
194 void ForEach(FunctionRef<void(const FilePath& path)> ref);
195
196 // Returns the next file or an empty string if there are no more results.
197 //
198 // The returned path will incorporate the |root_path| passed in the
199 // constructor: "<root_path>/file_name.txt". If the |root_path| is absolute,
200 // then so will be the result of Next().
201 FilePath Next();
202
203 // Returns info about the file last returned by Next(). Note that on Windows
204 // and Fuchsia, GetInfo() does not play well with INCLUDE_DOT_DOT. In
205 // particular, the GetLastModifiedTime() for the .. directory is 1601-01-01
206 // on Fuchsia (https://crbug.com/1106172) and is equal to the last modified
207 // time of the current directory on Windows (https://crbug.com/1119546).
208 // Must not be used with FileType::NAMES_ONLY.
209 FileInfo GetInfo() const;
210
211 // Once |Next()| returns an empty path, enumeration has been terminated. If
212 // termination was normal (i.e. no more results to enumerate) or ErrorPolicy
213 // is set to IGNORE_ERRORS, this returns FILE_OK. Otherwise it returns an
214 // error code reflecting why enumeration was stopped early.
215 File::Error GetError() const { return error_; }
216
217 private:
218 // Returns true if the given path should be skipped in enumeration.
219 bool ShouldSkip(const FilePath& path);
220
221 bool IsTypeMatched(bool is_dir) const;
222
223 bool IsPatternMatched(const FilePath& src) const;
224
225#if BUILDFLAG(IS_WIN)
226 const WIN32_FIND_DATA& find_data() const {
227 return *ChromeToWindowsType(&find_data_);
228 }
229
230 // True when find_data_ is valid.
231 bool has_find_data_ = false;
232 CHROME_WIN32_FIND_DATA find_data_;
233 HANDLE find_handle_ = INVALID_HANDLE_VALUE;
234
235#elif BUILDFLAG(IS_POSIX) || BUILDFLAG(IS_FUCHSIA)
236 // Marks the given inode as visited. Returns true if it is the first time that
237 // it got marked as visited.
238 bool MarkVisited(const stat_wrapper_t& st) {
239 return visited_[st.st_dev].insert(st.st_ino).second;
240 }
241
242 // The files in the current directory
243 std::vector<FileInfo> directory_entries_;
244
245#if BUILDFLAG(IS_ANDROID)
246 // The Android NDK (r23) does not declare `st_dev` as a `dev_t`, nor `st_ino`
247 // as an `ino_t`, hence the need for these decltypes.
248 using dev_t = decltype(stat_wrapper_t::st_dev);
249 using ino_t = decltype(stat_wrapper_t::st_ino);
250#endif
251
252 // Set of visited directories. Used to prevent infinite looping along circular
253 // symlinks and bind-mounts.
254 std::unordered_map<dev_t, std::unordered_set<ino_t>> visited_;
255
256 // The next entry to use from the directory_entries_ vector
257 size_t current_directory_entry_;
258#endif
259 FilePath root_path_;
260 const bool recursive_;
261 int file_type_;
262 FilePath::StringType pattern_;
263 const FolderSearchPolicy folder_search_policy_;
264 const ErrorPolicy error_policy_;
265 File::Error error_ = File::FILE_OK;
266
267 // A stack that keeps track of which subdirectories we still need to
268 // enumerate in the breadth-first search.
269 kiwi::stack<FilePath> pending_paths_;
270#if BUILDFLAG(IS_ANDROID)
271 // Matches pending_paths_, but with display names.
272 kiwi::stack<std::vector<std::string>> pending_subdirs_;
273 // Display names of subdirs of the current entry.
274 std::vector<std::string> subdirs_;
275#endif
276};
