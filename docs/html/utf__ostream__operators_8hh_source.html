21namespace std {
22// These functions are provided as a convenience for logging, which is where we
23// use streams (it is against Google style to use streams in other places). It
24// is designed to allow you to emit non-ASCII Unicode strings to the log file,
25// which is normally ASCII. It is relatively slow, so try not to use it for
26// common cases. Non-ASCII characters will be converted to UTF-8 by these
27// operators.
28//
29// The `std::basic_string<T>` overloads are necessary to allow logging types
30// which are implicitly convertible to `std::basic_string<T>`. Simply taking
31// `std::basic_string_view<T>` would not work because C++ only allows one
32// implicit conversion.
33BASE_EXPORT std::ostream& operator<<(std::ostream& out, const wchar_t* wstr);
34BASE_EXPORT std::ostream& operator<<(std::ostream& out, std::wstring_view wstr);
35BASE_EXPORT std::ostream& operator<<(std::ostream& out,
36 const std::wstring& wstr);
37
38BASE_EXPORT std::ostream& operator<<(std::ostream& out, const char16_t* str16);
39BASE_EXPORT std::ostream& operator<<(std::ostream& out,
40 std::u16string_view str16);
41BASE_EXPORT std::ostream& operator<<(std::ostream& out,
42 const std::u16string& str16);
43} // namespace std
