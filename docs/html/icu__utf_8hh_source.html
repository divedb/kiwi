249#define CBU8_INTERNAL_NEXT_OR_SUB(s, i, length, c, sub) \
250 CBUPRV_BLOCK_MACRO_BEGIN { \
251 (c) = (uint8_t)(s)[(i)++]; \
252 if (!CBU8_IS_SINGLE(c)) { \
253 uint8_t __t = 0; \
254 if ((i) != (length) && /* fetch/validate/assemble all but last trail \
255 byte */ \
256 ((c) >= 0xe0 \
257 ? ((c) < 0xf0 ? /* U+0800..U+FFFF except surrogates */ \
258 CBU8_LEAD3_T1_BITS[(c) &= 0xf] & \
259 (1 << ((__t = (s)[i]) >> 5)) && \
260 (__t &= 0x3f, 1) \
261 : /* U+10000..U+10FFFF */ \
262 ((c) -= 0xf0) <= 4 && \
263 CBU8_LEAD4_T1_BITS[(__t = (s)[i]) >> 4] & \
264 (1 << (c)) && \
265 ((c) = ((c) << 6) | (__t & 0x3f), \
266 ++(i) != (length)) && \
267 (__t = (s)[i] - 0x80) <= \
268 0x3f) && /* valid second-to-last trail byte */ \
269 ((c) = ((c) << 6) | __t, ++(i) != (length)) \
270 : /* U+0080..U+07FF */ \
271 (c) >= 0xc2 && ((c) &= 0x1f, 1)) && /* last trail byte */ \
272 (__t = (s)[i] - 0x80) <= 0x3f && \
273 ((c) = ((c) << 6) | __t, ++(i), 1)) { \
274 } else { \
275 (c) = (sub); /* ill-formed*/ \
276 } \
277 } \
278 } \
279 CBUPRV_BLOCK_MACRO_END
