162 // Returns the maximum time delta, which should be greater than any reasonable
163 // time delta we might compare it to. If converted to double with ToDouble()
164 // it becomes an IEEE double infinity. Use FiniteMax() if you want a very
165 // large number that doesn't do this. TimeDelta math saturates at the end
166 // points so adding to TimeDelta::Max() leaves the value unchanged.
167 // Subtracting should leave the value unchanged but currently changes it
168 // TODO(crbug.com/41405098).
169 static constexpr TimeDelta Max();
170
171 // Returns the minimum time delta, which should be less than than any
172 // reasonable time delta we might compare it to. For more details see the
173 // comments for Max().
174 static constexpr TimeDelta Min();

176 // Returns the maximum time delta which is not equivalent to infinity. Only
177 // subtracting a finite time delta from this time delta has a defined result.
178 static constexpr TimeDelta FiniteMax();

181 // adding a finite time delta to this time delta has a defined result.
182 static constexpr TimeDelta FiniteMin();

184 // Returns the internal numeric value of the TimeDelta object. Please don't
185 // use this and do arithmetic on it, as it is more error prone than using the

283 // This does floating-point division. For an integer result, either call
284 // IntDiv(), or (possibly clearer) use this operator with
285 // base::Clamp{Ceil,Floor,Round}() or base::saturated_cast() (for truncation).
286 // Note that converting to double here drops precision to 53 bits.
287 constexpr double operator/(TimeDelta a) const {
288 // 0/0 and inf/inf (any combination of positive and negative) are invalid
289 // (they are almost certainly not intentional, and result in NaN, which
290 // turns into 0 if clamped to an integer; this makes introducing subtle bugs
291 // too easy).
292 assert(!is_zero() || !a.is_zero());
293 assert(!is_inf() || !a.is_inf());
294
295 return ToDouble() / a.ToDouble();
296 }
297
298 constexpr int64_t IntDiv(TimeDelta a) const {
299 if (!is_inf() && !a.is_zero()) {
300 return int64_t{delta_ / a.delta_};
301 }
302
303 // For consistency, use the same edge case CHECKs and behavior as the code
304 // above.
305 assert(!is_zero() || !a.is_zero());
306 assert(!is_inf() || !a.is_inf());
307
308 return ((delta_ < 0) == (a.delta_ < 0))
309 ? std::numeric_limits<int64_t>::max()
310 : std::numeric_limits<int64_t>::min();
311 }
312
313 constexpr TimeDelta operator%(TimeDelta a) const {
314 return TimeDelta(
315 (is_inf() || a.is_zero() || a.is_inf()) ? delta_ : (delta_ % a.delta_));
316 }
317
318 constexpr TimeDelta& operator%=(TimeDelta other) {
319 return *this = (*this % other);
320 }

334 // Constructs a delta given the duration in microseconds. This is private
335 // to avoid confusion by callers with an integer constructor. Use
336 // base::Seconds, base::Milliseconds, etc. instead.
337 constexpr explicit TimeDelta(int64_t delta_us) : delta_(delta_us) {}
338 constexpr explicit TimeDelta(ClampedNumeric<int64_t> delta_us)
339 : delta_(delta_us) {}
340
341 // Returns a double representation of this TimeDelta's tick count. In
342 // particular, Max()/Min() are converted to +/-infinity.
343 constexpr double ToDouble() const {
344 if (!is_inf()) {
345 return static_cast<double>(delta_);
346 }
347
348 return (delta_ < 0) ? -std::numeric_limits<double>::infinity()
349 : std::numeric_limits<double>::infinity();
350 }
351
353 ClampedNumeric<int64_t> delta_ = 0;
354};

356constexpr TimeDelta TimeDelta::operator+(TimeDelta other) const {
357 if (!other.is_inf()) {
358 return TimeDelta(delta_ + other.delta_);
359 }
360
361 // Additions involving two infinities are only valid if signs match.
362 assert(!is_inf() || (delta_ == other.delta_));
363
364 return other;
365}
366
367constexpr TimeDelta TimeDelta::operator-(TimeDelta other) const {
368 if (!other.is_inf()) {
369 return TimeDelta(delta_ - other.delta_);
370 }
371
372 // Subtractions involving two infinities are only valid if signs differ.
373 CHECK_NE(int64_t{delta_}, int64_t{other.delta_});
374 return (other.delta_ < 0) ? Max() : Min();
375}

392// Provides value storage and comparison/math operations common to all time
393// classes. Each subclass provides for strong type-checking to ensure
394// semantically meaningful comparison/math of time values from the same clock
395// source or timeline.

397class TimeBase {
398 public:
399 static constexpr int64_t kHoursPerDay = 24;
400 static constexpr int64_t kSecondsPerMinute = 60;
401 static constexpr int64_t kMinutesPerHour = 60;
402 static constexpr int64_t kSecondsPerHour =
403 kSecondsPerMinute * kMinutesPerHour;
404 static constexpr int64_t kMillisecondsPerSecond = 1000;
405 static constexpr int64_t kMillisecondsPerDay =
406 kMillisecondsPerSecond * kSecondsPerHour * kHoursPerDay;
407 static constexpr int64_t kMicrosecondsPerMillisecond = 1000;
408 static constexpr int64_t kMicrosecondsPerSecond =
409 kMicrosecondsPerMillisecond * kMillisecondsPerSecond;
410 static constexpr int64_t kMicrosecondsPerMinute =
411 kMicrosecondsPerSecond * kSecondsPerMinute;
412 static constexpr int64_t kMicrosecondsPerHour =
413 kMicrosecondsPerMinute * kMinutesPerHour;
414 static constexpr int64_t kMicrosecondsPerDay =
415 kMicrosecondsPerHour * kHoursPerDay;
416 static constexpr int64_t kMicrosecondsPerWeek = kMicrosecondsPerDay * 7;
417 static constexpr int64_t kNanosecondsPerMicrosecond = 1000;
418 static constexpr int64_t kNanosecondsPerMillisecond =
419 kNanosecondsPerMicrosecond * kMicrosecondsPerMillisecond;
420 static constexpr int64_t kNanosecondsPerSecond =
421 kNanosecondsPerMicrosecond * kMicrosecondsPerSecond;
422
423 // TODO(crbug.com/40247732): Remove concept of "null" from base::Time.
424 //
425 // Warning: Be careful when writing code that performs math on time values,
426 // since it's possible to produce a valid "zero" result that should not be
427 // interpreted as a "null" value. If you find yourself using this method or
428 // the zero-arg default constructor, please consider using an optional to
429 // express the null state.
430 //
431 // Returns true if this object has not been initialized (probably).
432 constexpr bool is_null() const { return us_ == 0; }
433
434 // Returns true if this object represents the maximum/minimum time.
435 constexpr bool is_max() const { return *this == Max(); }
436 constexpr bool is_min() const { return *this == Min(); }
437 constexpr bool is_inf() const { return is_min() || is_max(); }

439 // Returns the maximum/minimum times, which should be non-null and
440 // greater/less than than any reasonable time with which we might compare it.
441 static constexpr TimeClass Max() {
442 return TimeClass(std::numeric_limits<int64_t>::max());
443 }
444
445 static constexpr TimeClass Min() {
446 return TimeClass(std::numeric_limits<int64_t>::min());
447 }
448
449 // For legacy serialization only. When serializing to `base::Value`, prefer

451 // `Time::ToDeltaSinceWindowsEpoch()` for `Time` and
452 // `TimeDelta::InMicroseconds()` for `TimeDelta`. See http://crbug.com/634507.
453 constexpr int64_t ToInternalValue() const { return us_; }
454
455 // The amount of time since the origin (or "zero") point. This is a syntactic
456 // convenience to aid in code readability, mainly for debugging/testing use
457 // cases.
458 //
459 // Warning: While the Time subclass has a fixed origin point, the origin for
460 // the other subclasses can vary each time the application is restarted.
461 constexpr TimeDelta since_origin() const;
462
463 // Compute the difference between two times.
464#if !defined(__aarch64__) && BUILDFLAG(IS_ANDROID)
465 NOINLINE // https://crbug.com/1369775
466#endif
467 constexpr TimeDelta
468 operator-(const TimeBase<TimeClass>& other) const;
469
470 // Return a new time modified by some delta.
471 constexpr TimeClass operator+(TimeDelta delta) const;
472 constexpr TimeClass operator-(TimeDelta delta) const;
473
474 // Modify by some time delta.
475 constexpr TimeClass& operator+=(TimeDelta delta) {
476 return static_cast<TimeClass&>(*this = (*this + delta));
477 }
478 constexpr TimeClass& operator-=(TimeDelta delta) {
479 return static_cast<TimeClass&>(*this = (*this - delta));
480 }
481
482 // Comparison operators
483 friend constexpr bool operator==(const TimeBase&, const TimeBase&) = default;
484 friend constexpr std::strong_ordering operator<=>(const TimeBase&,
485 const TimeBase&) = default;
486
487 protected:
488 constexpr explicit TimeBase(int64_t us) : us_(us) {}
489
490 // Time value in a microsecond timebase.
491 ClampedNumeric<int64_t> us_;
492};

518// Represents a wall clock time in UTC. Values are not guaranteed to be
519// monotonically non-decreasing and are subject to large amounts of skew.
520// Time is stored internally as microseconds since the Windows epoch (1601).
521class BASE_EXPORT Time : public time_internal::TimeBase<Time> {
522 public:
523 // Offset of UNIX epoch (1970-01-01 00:00:00 UTC) from Windows FILETIME epoch
524 // (1601-01-01 00:00:00 UTC), in microseconds. This value is derived from the
525 // following: ((1970-1601)*365+89)*24*60*60*1000*1000, where 89 is the number
526 // of leap year days between 1601 and 1970: (1970-1601)/4 excluding 1700,
527 // 1800, and 1900.
528 static constexpr int64_t kTimeTToMicrosecondsOffset =
529 INT64_C(11644473600000000);
530
531#if BUILDFLAG(IS_WIN)
532 // To avoid overflow in QPC to Microseconds calculations, since we multiply
533 // by kMicrosecondsPerSecond, then the QPC value should not exceed
534 // (2^63 - 1) / 1E6. If it exceeds that threshold, we divide then multiply.
535 static constexpr int64_t kQPCOverflowThreshold = INT64_C(0x8637BD05AF7);
536#endif
537
538// kExplodedMinYear and kExplodedMaxYear define the platform-specific limits
539// for values passed to FromUTCExploded() and FromLocalExploded(). Those
540// functions will return false if passed values outside these limits. The limits
541// are inclusive, meaning that the API should support all dates within a given
542// limit year.
543//
544// WARNING: These are not the same limits for the inverse functionality,
545// UTCExplode() and LocalExplode(). See method comments for further details.
546#if BUILDFLAG(IS_WIN)
547 static constexpr int kExplodedMinYear = 1601;
548 static constexpr int kExplodedMaxYear = 30827;
549#elif BUILDFLAG(IS_IOS) && !__LP64__
550 static constexpr int kExplodedMinYear = std::numeric_limits<int>::min();
551 static constexpr int kExplodedMaxYear = std::numeric_limits<int>::max();
552#elif BUILDFLAG(IS_APPLE)
553 static constexpr int kExplodedMinYear = 1902;
554 static constexpr int kExplodedMaxYear = std::numeric_limits<int>::max();
555#elif BUILDFLAG(IS_ANDROID)
556 // Though we use 64-bit time APIs on both 32 and 64 bit Android, some OS
557 // versions like KitKat (ARM but not x86 emulator) can't handle some early
558 // dates (e.g. before 1170). So we set min conservatively here.
559 static constexpr int kExplodedMinYear = 1902;
560 static constexpr int kExplodedMaxYear = std::numeric_limits<int>::max();
561#else
562 static constexpr int kExplodedMinYear =
563 (sizeof(time_t) == 4 ? 1902 : std::numeric_limits<int>::min());
564 static constexpr int kExplodedMaxYear =
565 (sizeof(time_t) == 4 ? 2037 : std::numeric_limits<int>::max());
566#endif

568 // Represents an exploded time. This is kind of like the Win32 SYSTEMTIME
569 // structure or the Unix "struct tm" with a few additions and changes to
570 // prevent errors.
571 //
572 // This structure always represents dates in the Gregorian calendar and always
573 // encodes day_of_week as Sunday==0, Monday==1, .., Saturday==6. This means
574 // that base::Time::LocalExplode and base::Time::FromLocalExploded only
575 // respect the current local time zone in the conversion and do *not* use a
576 // calendar or day-of-week encoding from the current locale.
577 //
578 // NOTE: Generally, you should prefer the functions in
579 // base/i18n/time_formatting.h (in particular,
580 // `UnlocalizedTimeFormatWithPattern()`) over trying to create a formatted
581 // time string from this object.
582 struct BASE_EXPORT Exploded {
583 int year; // Four digit year "2007"
584 int month; // 1-based month (values 1 = January, etc.)
585 int day_of_week; // 0-based day of week (0 = Sunday, etc.)
586 int day_of_month; // 1-based day of month (1-31)
587 int hour; // Hour within the current day (0-23)
588 int minute; // Minute within the current hour (0-59)
589 int second; // Second within the current minute (0-59 plus leap
590 // seconds which may take it up to 60).
591 int millisecond; // Milliseconds within the current second (0-999)
592
593 // A cursory test for whether the data members are within their
594 // respective ranges. A 'true' return value does not guarantee the
595 // Exploded value can be successfully converted to a Time value.
596 bool HasValidValues() const;
597 };

669 // Converts to/from a number of milliseconds since the Unix epoch.
670 // TODO(crbug.com/40286584): Add ...PreservingNull() versions; see comments in
671 // the implementation of FromMillisecondsSinceUnixEpoch().
672 static constexpr Time FromMillisecondsSinceUnixEpoch(int64_t dt);
673 static constexpr Time FromMillisecondsSinceUnixEpoch(double dt);
674 // Explicitly forward calls with smaller integral types to the int64_t
675 // version; otherwise such calls would need to manually cast their args to
676 // int64_t, since the compiler isn't sure whether to promote to int64_t or
677 // double.
678 template <typename T>
679 requires(std::integral<T> && !std::same_as<T, int64_t> &&
680 (sizeof(T) < sizeof(int64_t) ||
681 (sizeof(T) == sizeof(int64_t) && std::is_signed_v<T>)))
682 static constexpr Time FromMillisecondsSinceUnixEpoch(T ms_since_epoch) {
683 return FromMillisecondsSinceUnixEpoch(int64_t{ms_since_epoch});
684 }
685 constexpr int64_t InMillisecondsSinceUnixEpoch() const;
686 // Don't use InMillisecondsFSinceUnixEpoch() in new code, since it contains a

1328// Represents a thread-specific clock that runs only while the thread is
1329// scheduled. This has the effect of counting time spent actually executing
1330// code, but not time spent blocked (e.g. on I/O), or ready and waiting to be
1331// run.
1332//
1333// Note: This is typically significantly more expensive than TimeTicks. For
1334// instance, on Linux-based systems, it requires a true system call, whereas
1335// TimeTicks::Now() calls are usually handled through the vDSO. This does not
1336// matter if a couple us of overhead is not important to you, but do not call
1337// this in a tight loop, or for sub-microsecond intervals.
1338//
1339// For instance, in 2024 on a Linux system, in a simple loop:
1340// - TimeTicks::Now() takes 27ns per loop iteration
1341// - ThreadTicks::Now() takes 875ns per loop iteration. Actual cost is likely
1342// higher in Chromium due to the sandbox (seccomp-BPF).

1351 return true;
1352#elif BUILDFLAG(IS_WIN)

1354#else
1355 return false;
1356#endif
