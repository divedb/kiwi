36template <typename Range>
37class RangeOfRvaluesAdapter {
38 public:
39 explicit RangeOfRvaluesAdapter(Range&& range KIWI_LIFETIME_BOUND)
40 : range_(std::forward<Range>(range)) {}
41 RangeOfRvaluesAdapter(const RangeOfRvaluesAdapter&) = default;
42 RangeOfRvaluesAdapter& operator=(const RangeOfRvaluesAdapter&) = delete;
43
44 auto size() const
45 requires std::ranges::sized_range<Range>
46 {
47 return std::ranges::size(range_);
48 }
49
50 auto begin() { return std::make_move_iterator(std::ranges::begin(range_)); }
51 auto end() { return std::make_move_iterator(std::ranges::end(range_)); }
52
53 private:
54 // RAW_PTR_EXCLUSION: References a STACK_ALLOCATED class. It is only used
55 // inside for loops. Ideally, the container being iterated over should be the
56 // one held via a raw_ref/raw_ptrs.
57 Range&& range_;
58};

60// Internal adapter class for implementing kiwi::Reversed.
61// TODO(crbug.com/378623811): Parts of this (e.g. the `size()` helper) should be
62// extracted to a base template that can be shared/reused. In addition, this
63// should be constrained to Ts that satisfy the std::ranges::range concept.
64template <typename Range>
65class ReversedAdapter {
66 public:
67 explicit ReversedAdapter(Range&& range KIWI_LIFETIME_BOUND)
68 : range_(std::forward<Range>(range)) {}
69 ReversedAdapter(const ReversedAdapter&) = default;
70 ReversedAdapter& operator=(const ReversedAdapter&) = delete;
71
72 auto begin() { return std::rbegin(range_); }
73 auto begin() const { return std::rbegin(range_); }
74 auto cbegin() const { return std::crbegin(range_); }
75
76 auto end() { return std::rend(range_); }
77 auto end() const { return std::rend(range_); }
78 auto cend() const { return std::crend(range_); }
79
80 auto size() const
81 requires std::ranges::sized_range<Range>
82 {
83 return std::ranges::size(range_);
84 }
85
86 private:
87 // RAW_PTR_EXCLUSION: References a STACK_ALLOCATED class. It is only used
88 // inside for loops. Ideally, the container being iterated over should be the
89 // one held via a raw_ref/raw_ptrs.
90 Range&& range_;
91};
