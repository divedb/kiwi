159 [[nodiscard]] int compare(StringRef rhs) const {
160 // Check the prefix for a mismatch.
161 if (int res =
162 compare_memory(data_, rhs.data_, std::min(length_, rhs.length_))) {
163 return res < 0 ? -1 : 1;
164 }
165
167 if (length_ == rhs.length_) {
168 return 0;
169 }
170
171 return length_ < rhs.length_ ? -1 : 1;

292 [[nodiscard]] size_t find_if(function_ref<bool(char)> f,
293 size_t from = 0) const {
294 StringRef s = drop_front(from);
295 while (!s.empty()) {
296 if (f(s.front())) {
297 return size() - s.size();
298 }
299
300 s = s.drop_front();
301 }
302
303 return npos;
304 }

721 [[nodiscard]] StringRef detectEOL() const {
722 size_t Pos = find('\r');
723 if (Pos == npos) {
724 // If there is no carriage return, assume unix
725 return "\n";
726 }
727 if (Pos + 1 < Length && Data[Pos + 1] == '\n') return "\r\n"; // Windows
728 if (Pos > 0 && Data[Pos - 1] == '\n') return "\n\r"; // You monster!
729 return "\r"; // Classic Mac
