17// Default to regular `std::basic_string::resize()`.
18template <typename CharT>
19void Resize(std::basic_string<CharT>& str, size_t total_size) {
20 str.resize(total_size);
21}

23// Optimized version of `std::basic_string::resize()` that skips zero
24// initialization of appended characters. Reading from the newly allocated
25// characters results in undefined behavior if they are not explicitly
26// initialized afterwards. Available in C++23 as
27// `std::basic_string::resize_and_overwrite()`:
28// https://en.cppreference.com/w/cpp/string/basic_string/resize_and_overwrite
29template <typename CharT>
30 requires requires(std::basic_string<CharT>& str, size_t total_size) {
31 { str.__resize_default_init(total_size) } -> std::same_as<void>;
32 }
33auto Resize(std::basic_string<CharT>& str, size_t total_size) {
34 str.__resize_default_init(total_size);
35}

37// Appends `pieces` to `dest`. Instead of simply calling `dest.append()`
38// `pieces.size()` times, this method first resizes `dest` to be of the desired
39// size, and then appends each piece via `std::char_traits::copy`. This achieves
40// two goals:
41// 1) Allocating the desired size all at once avoids other allocations that
42// could happen if intermediate allocations did not reserve enough capacity.
43// 2) Invoking std::char_traits::copy instead of std::basic_string::append
44// avoids having to write the terminating '\0' character n times.
45template <typename CharT, typename StringT>
46void StrAppendT(std::basic_string<CharT>& dest, span<const StringT> pieces) {
47 const size_t initial_size = dest.size();
48 size_t total_size = initial_size;
49 for (const auto& cur : pieces) {
50 total_size += cur.size();
51 }
52
53 // Note: As opposed to `reserve()` calling `resize()` with an argument smaller
54 // than the current `capacity()` does not result in the string releasing spare
55 // capacity. Furthermore, common std::string implementations apply a geometric
56 // growth strategy if the current capacity is not sufficient for the newly
57 // added characters. Since this codepath is also triggered by `resize()`, we
58 // don't have to manage the std::string's capacity ourselves here to avoid
59 // performance hits in case `StrAppend()` gets called in a loop.
60 Resize(dest, total_size);
61 CharT* dest_char = &dest[initial_size];
62 for (const auto& cur : pieces) {
63 std::char_traits<CharT>::copy(dest_char, cur.data(), cur.size());
64 UNSAFE_TODO(dest_char += cur.size());
65 }
66}
67
68template <typename StringT>
69auto StrCatT(span<const StringT> pieces) {
70 std::basic_string<typename StringT::value_type> result;
71 StrAppendT(result, pieces);
72 return result;
73}
