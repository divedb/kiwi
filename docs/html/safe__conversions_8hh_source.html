49// Signed to signed range comparison.
50template <typename Dst, typename Src>
51 requires(std::signed_integral<Dst> && std::signed_integral<Src> &&
52 !kIsTypeInRangeForNumericType<Dst, Src>)
53struct IsValueInRangeFastOp<Dst, Src> {
54 static constexpr bool is_supported = true;
55
56 static constexpr bool Do(Src value) {
57 // Just downcast to the smaller type, sign extend it back to the original
58 // type, and then see if it matches the original value.
59 return value == static_cast<Dst>(value);
60 }
61};
62
63// Signed to unsigned range comparison.
64template <typename Dst, typename Src>
65 requires(std::unsigned_integral<Dst> && std::signed_integral<Src> &&
66 !kIsTypeInRangeForNumericType<Dst, Src>)
67struct IsValueInRangeFastOp<Dst, Src> {
68 static constexpr bool is_supported = true;
69
70 static constexpr bool Do(Src value) {
71 // We cast a signed as unsigned to overflow negative values to the top,
72 // then compare against whichever maximum is smaller, as our upper bound.
73 return as_unsigned(value) <= as_unsigned(kCommonMax<Src, Dst>);
74 }
75};

77// Convenience function that returns true if the supplied value is in range
78// for the destination type.
79template <typename Dst, typename Src>
80 requires(IsNumeric<Src> && std::is_arithmetic_v<Dst> &&
81 std::numeric_limits<Dst>::lowest() < std::numeric_limits<Dst>::max())
82constexpr bool IsValueInRangeForNumericType(Src value) {
83 using SrcType = UnderlyingType<Src>;
84 const auto underlying_value = static_cast<SrcType>(value);
85 return internal::IsValueInRangeFastOp<Dst, SrcType>::is_supported
86 ? internal::IsValueInRangeFastOp<Dst, SrcType>::Do(
87 underlying_value)
88 : internal::DstRangeRelationToSrcRange<Dst>(underlying_value)
89 .IsValid();
90}

92// checked_cast<> is analogous to static_cast<> for numeric types,
93// except that it CHECKs that the specified numeric conversion will not
94// overflow or underflow. NaN source will always trigger a CHECK.
95template <typename Dst, class CheckHandler = internal::CheckOnFailure,
96 typename Src>
97 requires(IsNumeric<Src> && std::is_arithmetic_v<Dst> &&
98 std::numeric_limits<Dst>::lowest() < std::numeric_limits<Dst>::max())
99constexpr Dst checked_cast(Src value) {
100 // This throws a compile-time error on evaluating the constexpr if it can be
101 // determined at compile-time as failing, otherwise it will CHECK at runtime.
102 if (IsValueInRangeForNumericType<Dst>(value)) [[likely]] {
103 return static_cast<Dst>(static_cast<UnderlyingType<Src>>(value));
104 }
105 return CheckHandler::template HandleFailure<Dst>();
106}

108// Default boundaries for integral/float: max/infinity, lowest/-infinity, 0/NaN.
109// You may provide your own limits (e.g. to saturated_cast) so long as you
110// implement all of the static constexpr member functions in the class below.
111template <typename T>
112struct SaturationDefaultLimits : public std::numeric_limits<T> {
113 static constexpr T NaN() {
114 if constexpr (std::numeric_limits<T>::has_quiet_NaN) {
115 return std::numeric_limits<T>::quiet_NaN();
116 } else {
117 return T();
118 }
119 }
120 using std::numeric_limits<T>::max;
121 static constexpr T Overflow() {
122 if constexpr (std::numeric_limits<T>::has_infinity) {
123 return std::numeric_limits<T>::infinity();
124 } else {
125 return std::numeric_limits<T>::max();
126 }
127 }
128 using std::numeric_limits<T>::lowest;
129 static constexpr T Underflow() {
130 if constexpr (std::numeric_limits<T>::has_infinity) {
131 return std::numeric_limits<T>::infinity() * -1;
132 } else {
133 return std::numeric_limits<T>::lowest();
134 }
135 }
136};

138template <typename Dst, template <typename> class S, typename Src>
139constexpr Dst saturated_cast_impl(Src value, RangeCheck constraint) {
140 // For some reason clang generates much better code when the branch is
141 // structured exactly this way, rather than a sequence of checks.
142 return !constraint.IsOverflowFlagSet()
143 ? (!constraint.IsUnderflowFlagSet() ? static_cast<Dst>(value)
144 : S<Dst>::Underflow())
145 // Skip this check for integral Src, which cannot be NaN.
146 : (std::is_integral_v<Src> || !constraint.IsUnderflowFlagSet()
147 ? S<Dst>::Overflow()
148 : S<Dst>::NaN());
149}

173template <typename Dst, typename Src>
174 requires(std::integral<Src> && std::integral<Dst> &&
175 !SaturateFastAsmOp<Dst, Src>::is_supported)
176struct SaturateFastOp<Dst, Src> {
177 static constexpr bool is_supported = true;
178 static constexpr Dst Do(Src value) {
179 // The exact order of the following is structured to hit the correct
180 // optimization heuristics across compilers. Do not change without
181 // checking the emitted code.
182 const Dst saturated = CommonMaxOrMin<Dst, Src>(
183 kIsMaxInRangeForNumericType<Dst, Src> ||
184 (!kIsMinInRangeForNumericType<Dst, Src> && IsValueNegative(value)));
185 if (IsValueInRangeForNumericType<Dst>(value)) [[likely]] {
186 return static_cast<Dst>(value);
187 }
188 return saturated;
189 }
190};

192// saturated_cast<> is analogous to static_cast<> for numeric types, except
193// that the specified numeric conversion will saturate by default rather than
194// overflow or underflow, and NaN assignment to an integral will return 0.
195// All boundary condition behaviors can be overridden with a custom handler.
196template <typename Dst,
197 template <typename> class SaturationHandler = SaturationDefaultLimits,
198 typename Src>
199constexpr Dst saturated_cast(Src value) {
200 using SrcType = UnderlyingType<Src>;
201 const auto underlying_value = static_cast<SrcType>(value);
202 return !std::is_constant_evaluated() &&
203 SaturateFastOp<Dst, SrcType>::is_supported &&
204 std::is_same_v<SaturationHandler<Dst>,
205 SaturationDefaultLimits<Dst>>
206 ? SaturateFastOp<Dst, SrcType>::Do(underlying_value)
207 : saturated_cast_impl<Dst, SaturationHandler, SrcType>(
208 underlying_value,
209 DstRangeRelationToSrcRange<Dst, SaturationHandler, SrcType>(
210 underlying_value));
211}

213// strict_cast<> is analogous to static_cast<> for numeric types, except that
214// it will cause a compile failure if the destination type is not large enough
215// to contain any value in the source type. It performs no runtime checking.
216template <typename Dst, typename Src, typename SrcType = UnderlyingType<Src>>
217 requires(
218 IsNumeric<Src> && std::is_arithmetic_v<Dst> &&
219 // If you got here from a compiler error, it's because you tried to assign
220 // from a source type to a destination type that has insufficient range.
221 // The solution may be to change the destination type you're assigning to,
222 // and use one large enough to represent the source.
223 // Alternatively, you may be better served with the checked_cast<> or
224 // saturated_cast<> template functions for your particular use case.
225 kStaticDstRangeRelationToSrcRange<Dst, SrcType> ==
226 NumericRangeRepresentation::kContained)
227constexpr Dst strict_cast(Src value) {
228 return static_cast<Dst>(static_cast<SrcType>(value));

242// StrictNumeric implements compile time range checking between numeric types by
243// wrapping assignment operations in a strict_cast. This class is intended to be
244// used for function arguments and return types, to ensure the destination type
245// can always contain the source type. This is essentially the same as enforcing
246// -Wconversion in gcc and C4302 warnings on MSVC, but it can be applied
247// incrementally at API boundaries, making it easier to convert code so that it
248// compiles cleanly with truncation warnings enabled.
249// This template should introduce no runtime overhead, but it also provides no
250// runtime checking of any of the associated mathematical operations. Use
251// CheckedNumeric for runtime range checks of the actual value being assigned.
252template <typename T>
253 requires std::is_arithmetic_v<T>
254class StrictNumeric {
255 public:
256 using type = T;
257
258 constexpr StrictNumeric() : value_(0) {}
259
260 // Copy constructor.
261 template <typename Src>
262 constexpr StrictNumeric(const StrictNumeric<Src>& rhs)
263 : value_(strict_cast<T>(rhs.value_)) {}
264
265 // This is not an explicit constructor because we implicitly upgrade regular
266 // numerics to StrictNumerics to make them easier to use.
267 template <typename Src>
268 // NOLINTNEXTLINE(google-explicit-constructor)
269 constexpr StrictNumeric(Src value) : value_(strict_cast<T>(value)) {}

271 // If you got here from a compiler error, it's because you tried to assign
272 // from a source type to a destination type that has insufficient range.
273 // The solution may be to change the destination type you're assigning to,
274 // and use one large enough to represent the source.
275 // If you're assigning from a CheckedNumeric<> class, you may be able to use
276 // the AssignIfValid() member function, specify a narrower destination type to
277 // the member value functions (e.g. val.template ValueOrDie<Dst>()), use one
278 // of the value helper functions (e.g. ValueOrDieForType<Dst>(val)).
279 // If you've encountered an _ambiguous overload_ you can use a static_cast<>
280 // to explicitly cast the result to the destination type.
281 // If none of that works, you may be better served with the checked_cast<> or
282 // saturated_cast<> template functions for your particular use case.
283 template <typename Dst>
284 requires(kIsNumericRangeContained<Dst, T>)
285 constexpr operator Dst() const { // NOLINT(google-explicit-constructor)
286 return static_cast<ArithmeticOrUnderlyingEnum<Dst>>(value_);
287 }
288
289 // Unary negation does not require any conversions.
290 constexpr bool operator!() const { return !value_; }
291
292 private:
293 template <typename U>
294 requires std::is_arithmetic_v<U>
295 friend class StrictNumeric;
296
297 T value_;
298};

341// floating -> integral conversions that saturate and thus can actually return
342// an integral type.
343//
344// Generally, what you want is saturated_cast<Dst>(std::nearbyint(x)), which
345// rounds correctly according to IEEE-754 (round to nearest, ties go to nearest
346// even number; this avoids bias). If your code is performance-critical
347// and you are sure that you will never overflow, you can use std::lrint()
348// or std::llrint(), which return a long or long long directly.
349//
350// Below are convenience functions around similar patterns, except that
351// they round in nonstandard directions and will generally be slower.
352
353// Rounds towards negative infinity (i.e., down).
354template <typename Dst = int, typename Src>
355 requires(std::integral<Dst> && std::floating_point<Src>)
356Dst ClampFloor(Src value) {
357 return saturated_cast<Dst>(std::floor(value));
358}
359
360// Rounds towards positive infinity (i.e., up).
361template <typename Dst = int, typename Src>
362 requires(std::integral<Dst> && std::floating_point<Src>)
363Dst ClampCeil(Src value) {
364 return saturated_cast<Dst>(std::ceil(value));
365}

367// Rounds towards nearest integer, with ties away from zero.
368// This means that 0.5 will be rounded to 1 and 1.5 will be rounded to 2.
369// Similarly, -0.5 will be rounded to -1 and -1.5 will be rounded to -2.
370//
371// This is normally not what you want accuracy-wise (it introduces a small bias
372// away from zero), and it is not the fastest option, but it is frequently what
373// existing code expects. Compare with saturated_cast<Dst>(std::nearbyint(x))
374// or std::lrint(x), which would round 0.5 and -0.5 to 0 but 1.5 to 2 and
375// -1.5 to -2.
376template <typename Dst = int, typename Src>
377 requires(std::integral<Dst> && std::floating_point<Src>)
378Dst ClampRound(Src value) {
379 const Src rounded = std::round(value);
380 return saturated_cast<Dst>(rounded);
381}
