113// CFCast<>() and CFCastStrict<>() cast a basic CFTypeRef to a more specific
114// CoreFoundation type. The compatibility of the passed object is found by
115// comparing its opaque type against the requested type identifier. If the
116// supplied object is not compatible with the requested return type, CFCast<>()
117// returns null and CFCastStrict<>() will CHECK. Providing a null pointer to
118// either variant results in null being returned without triggering any CHECK.
119//
120// Example usage:
121// CFNumberRef some_number = base::apple::CFCast<CFNumberRef>(
122// CFArrayGetValueAtIndex(array, index));
123//
124// CFTypeRef hello = CFSTR("hello world");
125// CFStringRef some_string = base::apple::CFCastStrict<CFStringRef>(hello);
126
127template <typename T>
128T CFCast(const CFTypeRef& cf_val);
129
130template <typename T>
131T CFCastStrict(const CFTypeRef& cf_val);
132

136 \
137 template <> \

166// Helper function for GetValueFromDictionary to create the error message
167// that appears when a type mismatch is encountered.
168BASE_EXPORT std::string GetValueFromDictionaryErrorMessage(
169 CFStringRef key, const std::string& expected_type, CFTypeRef value);

171// Utility function to pull out a value from a dictionary, check its type, and
172// return it. Returns NULL if the key is not present or of the wrong type.
173template <typename T>
174T GetValueFromDictionary(CFDictionaryRef dict, CFStringRef key) {
175 CFTypeRef value = CFDictionaryGetValue(dict, key);
176 T value_specific = CFCast<T>(value);
177
178 if (value && !value_specific) {
179 std::string expected_type = TypeNameForCFType(value_specific);
180 DLOG(WARNING) << GetValueFromDictionaryErrorMessage(key, expected_type,
181 value);
182 }
183
184 return value_specific;
185}
