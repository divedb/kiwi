25// Function pointers implicitly convert to `bool`, so use this to avoid printing
26// function pointers as "true"/"false".
27template <typename T>
28concept WillBeIncorrectlyStreamedAsBool =
29 std::is_function_v<std::remove_pointer_t<T>>;
30
31// Fallback case when there is no better representation.
32template <typename T>
33struct ToStringHelper {
34 static void Stringify(const T& v, std::ostringstream& ss) {
35 // We cast to `void*` to avoid converting a char-like type to char-like*
36 // which operator<< treats as a string but does not support for multi-byte
37 // char-like types.
38 ss << "[" << sizeof(v) << "-byte object at 0x"
39 << static_cast<const void*>(std::addressof(v)) << "]";
40 }
41};
42
43// Boolean values. (Handled explicitly so as to not rely on the behavior of
44// std::boolalpha.)
45template <>
46struct ToStringHelper<bool> {
47 static void Stringify(const bool& v, std::ostringstream& ss) {
48 ss << (v ? "true" : "false");
49 }
50};
51
52// Most streamables.
53template <typename T>
54 requires(SupportsOstreamOperator<const T&> &&
55 !WillBeIncorrectlyStreamedAsBool<T>)
56struct ToStringHelper<T> {
57 static void Stringify(const T& v, std::ostringstream& ss) { ss << v; }
58};
59
60// Functions and function pointers.
61template <typename T>
62 requires(SupportsOstreamOperator<const T&> &&
63 WillBeIncorrectlyStreamedAsBool<T>)
64struct ToStringHelper<T> {
65 static void Stringify(const T& v, std::ostringstream& ss) {
66 ToStringHelper<const void*>::Stringify(reinterpret_cast<const void*>(v),
67 ss);
68 }
69};
70
71// Integral types that can't stream, like char16_t.
72template <typename T>
73 requires(!SupportsOstreamOperator<const T&> && std::is_integral_v<T>)
74struct ToStringHelper<T> {
75 static void Stringify(const T& v, std::ostringstream& ss) {
76 if constexpr (std::is_signed_v<T>) {
77 static_assert(sizeof(T) <= 8);
78 ss << static_cast<int64_t>(v);
79 } else {
80 static_assert(sizeof(T) <= 8);
81 ss << static_cast<uint64_t>(v);
82 }
83 }
84};
85
86// Non-streamables that have a `ToString` member.
87template <typename T>
88 requires(!SupportsOstreamOperator<const T&> && SupportsToString<const T&>)
89struct ToStringHelper<T> {
90 static void Stringify(const T& v, std::ostringstream& ss) {
91 // .ToString() may not return a std::string, e.g. blink::WTF::String.
92 ToStringHelper<decltype(v.ToString())>::Stringify(v.ToString(), ss);
93 }
94};
95
96// Non-streamable enums (i.e. scoped enums where no `operator<<` overload was
97// declared).
98template <typename T>
99 requires(!SupportsOstreamOperator<const T&> && std::is_enum_v<T>)
100struct ToStringHelper<T> {
101 static void Stringify(const T& v, std::ostringstream& ss) {
102 using UT = typename std::underlying_type_t<T>;
103 ToStringHelper<UT>::Stringify(static_cast<UT>(v), ss);
104 }
105};
106
107// Tuples. Will recursively apply `ToString()` to each value in the tuple.
108template <typename... T>
109struct ToStringHelper<std::tuple<T...>> {
110 template <size_t... I>
111 static void StringifyHelper(const std::tuple<T...>& values,
112 std::index_sequence<I...>,
113 std::ostringstream& ss) {
114 ss << "<";
115 (..., (ss << (I == 0 ? "" : ", "), ss << ToString(std::get<I>(values))));
116 ss << ">";
117 }
118
119 static void Stringify(const std::tuple<T...>& v, std::ostringstream& ss) {
120 StringifyHelper(v, std::make_index_sequence<sizeof...(T)>(), ss);
121 }
122};

124} // namespace internal

126// Converts any type to a string, preferring defined operator<<() or ToString()
127// methods if they exist.
128template <typename T>
129std::string ToString(const T& value) {
130 std::ostringstream ss;
131 internal::ToStringHelper<std::remove_cvref_t<decltype(value)>>::Stringify(
132 value, ss);
133 return ss.str();
134}
