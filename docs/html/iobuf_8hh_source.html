317 static inline uintptr_t PackFlagsAndSharedInfo(uintptr_t flags,
318 SharedInfo* info) noexcept {
319 uintptr_t uinfo = reinterpret_cast<uintptr_t>(info);
320
321 DCHECK_EQ(flags & ~kFlagMask, 0u);
322 DCHECK_EQ(uinfo & kFlagMask, 0u);
323
324 return flags | uinfo;
325 }

328 return reinterpret_cast<SharedInfo*>(flags_and_shared_info_ & ~kFlagMask);

331 inline void SetSharedInfo(SharedInfo* info) noexcept {
332 uintptr_t uinfo = reinterpret_cast<uintptr_t>(info);

335
336 flags_and_shared_info_ = (flags_and_shared_info_ & kFlagMask) | uinfo;

1026 void advance(std::size_t amount) {
1027 // In debug builds, assert if there is a problem.
1028 assert(amount <= tailroom());
1029
1030 if (length_ > 0) {
1031 memmove(data_ + amount, data_, length_);
1032 }
1033 data_ += amount;

1190 void reserve(std::size_t minHeadroom, std::size_t minTailroom) {
1191 // Maybe we don't need to do anything.
1192 if (headroom() >= minHeadroom && tailroom() >= minTailroom) {
1193 return;
1194 }
1195 // If the buffer is empty but we have enough total room (head + tail),
1196 // move the data_ pointer around.
1197 if (length() == 0 && headroom() + tailroom() >= minHeadroom + minTailroom) {
1198 data_ = writableBuffer() + minHeadroom;
1199 return;
1200 }
1201 // Bah, we have to do actual work.
1202 reserveSlow(minHeadroom, minTailroom);
1203 }

1461 template <typename Observer>
1462 bool appendSharedInfoObserver(Observer&& observer) {
1463 SharedInfo* info = sharedInfo();
1464 if (!info) {
1465 return false;
1466 }
1467
1468 auto* entry =
1469 new SharedInfoObserverEntry<Observer>(std::forward<Observer>(observer));
1470 std::lock_guard<MicroSpinLock> guard(info->observerListLock);
1471 if (!info->observerListHead) {
1472 info->observerListHead = entry;
1473 } else {
1474 // prepend
1475 entry->next = info->observerListHead;
1476 entry->prev = info->observerListHead->prev;
1477 info->observerListHead->prev->next = entry;
1478 info->observerListHead->prev = entry;
1479 }
1480
1481 return true;
1482 }
