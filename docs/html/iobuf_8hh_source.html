406 static inline uintptr_t PackFlagsAndSharedInfo(uintptr_t flags,
407 SharedInfo* info) noexcept {
408 uintptr_t uinfo = reinterpret_cast<uintptr_t>(info);
409
410 DCHECK_EQ(flags & ~kFlagMask, 0u);
411 DCHECK_EQ(uinfo & kFlagMask, 0u);
412
413 return flags | uinfo;
414 }

425 flags_and_shared_info_ = (flags_and_shared_info_ & kFlagMask) | uinfo;

1177 void reserve(std::size_t min_head_room, std::size_t min_tail_room) {
1178 // Maybe we don't need to do anything.
1179 if (HeadRoom() >= min_head_room && TailRoom() >= min_tail_room) {
1180 return;
1181 }
1182
1183 // If the buffer is empty but we have enough total room (head + tail),
1184 // move the data_ pointer around.
1185 if (Length() == 0 &&
1186 HeadRoom() + TailRoom() >= min_head_room + min_tail_room) {
1187 data_ = WritableBuffer() + min_head_room;
1188
1189 return;
1190 }
1191
1192 // Bah, we have to do actual work.
1193 ReserveSlow(min_head_room, min_tail_room);
1194 }

1429 template <typename Observer>
1430 bool AppendSharedInfoObserver(Observer&& observer) {
1431 auto info = GetSharedInfo();
1432
1433 if (!info) {
1434 return false;
1435 }
1436
1437 auto* entry =
1438 new SharedInfoObserverEntry<Observer>(std::forward<Observer>(observer));
1439 std::lock_guard<std::mutex> guard(info->observer_list_lock);
1440
1441 if (!info->observer_list_head) {
1442 info->observer_list_head = entry;
1443 } else {
1444 // prepend
1445 entry->next = info->observer_list_head;
1446 entry->prev = info->observer_list_head->prev;
1447 info->observer_list_head->prev->next = entry;
1448 info->observer_list_head->prev = entry;
1449 }
1450
1451 return true;
1452 }

2029class IOBuf::Iterator
2030 : public detail::IteratorFacade<IOBuf::Iterator, ByteRange const,
2031 std::forward_iterator_tag> {
2032 void SetVal() {
2033 auto tmp = const_cast<IOBuf*>(pos_);
2034
2035 val_ = ByteRange(tmp->WritableData(), tmp->WritableTail());
2036 }
2037
2038 void AdjustForEnd() {
2039 if (pos_ == end_) {
2040 pos_ = end_ = nullptr;
2041 val_ = ByteRange();
2042 } else {
2043 SetVal();
2044 }
2045 }
2046
2047 public:
2048 // Note that IOBufs are stored as a circular list without a guard node,
2049 // so pos == end is ambiguous (it may mean "begin" or "end"). To solve
2050 // the ambiguity (at the cost of one extra comparison in the "increment"
2051 // code path), we define end iterators as having pos_ == end_ == nullptr
2052 // and we only allow forward iteration.
2053 explicit Iterator(const IOBuf* pos, const IOBuf* end) : pos_(pos), end_(end) {
2054 // Sadly, we must return by const reference, not by value.
2055 if (pos_) {
2056 SetVal();
2057 }
2058 }
2059
2060 Iterator() = default;
2061 Iterator(Iterator const& rhs) : Iterator(rhs.pos_, rhs.end_) {}
2062 Iterator& operator=(Iterator const& rhs) {
2063 pos_ = rhs.pos_;
2064 end_ = rhs.end_;
2065
2066 if (pos_) {
2067 SetVal();
2068 }
2069
2070 return *this;
2071 }
2072
2073 const ByteRange& Dereference() const { return val_; }
2074 bool IsEqual(const Iterator& other) const {
2075 // We must compare end_ in addition to pos_, because forward traversal
2076 // requires that if two iterators are equal (a == b) and dereferenceable,
2077 // then ++a == ++b.
2078 return pos_ == other.pos_ && end_ == other.end_;
2079 }
2080
2081 void Increment() {
2082 pos_ = pos_->Next();
2083 AdjustForEnd();
2084 }
2085
2086 private:
2087 const IOBuf* pos_{nullptr};
2088 const IOBuf* end_{nullptr};
2089 ByteRange val_;
2090};
