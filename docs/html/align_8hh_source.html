57// True if it may be presumed that the platform has static extended alignment;
58// false if it may not be so presumed, even when the platform might actually
59// have it. Static extended alignment refers to extended alignment of objects
60// with automatic, static, or thread storage. Whether the there is support for
61// dynamic extended alignment is a property of the allocator which is used for
62// each given dynamic allocation.
63//
64// Currently, very heuristical - only non-mobile 64-bit linux gets the extended
65// alignment treatment. Theoretically, this could be tuned better.
66#if BUILDFLAG(IS_LINUX)
67inline constexpr bool kHasExtendedAlignment =
68 sizeof(void*) >= sizeof(std::uint64_t);
69#else
70inline constexpr bool kHasExtendedAlignment = false;
71#endif

77template <typename... Ts>
78struct max_align_t_ {
79 static constexpr std::size_t value() {
80 std::size_t const values[] = {0u, alignof(Ts)...};
81 std::size_t r = 0u;
82
83 for (auto const v : values) {
84 r = r < v ? v : r;
85 }
86
87 return r;
88 }
89};

91using max_align_v_ =
92 max_align_t_<long double, double, float, long long int, long int, int,
93 short int, bool, char, char16_t, char32_t, wchar_t, void*,
94 std::max_align_t>;

96} // namespace detail
