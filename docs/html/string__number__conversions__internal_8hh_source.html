23template <typename STR, typename INT>
24static STR IntToStringT(INT value) {
25 // log10(2) ~= 0.3 bytes needed per bit or per byte log10(2**8) ~= 2.4.
26 // So round up to allocate 3 output characters per byte, plus 1 for '-'.
27 const size_t kOutputBufSize =
28 3 * sizeof(INT) + std::numeric_limits<INT>::is_signed;
29
30 // Create the string in a temporary buffer, write it back to front, and
31 // then return the substr of what we ended up using.
32 using CHR = typename STR::value_type;
33 CHR outbuf[kOutputBufSize];
34
35 // The ValueOrDie call below can never fail, because UnsignedAbs is valid
36 // for all valid inputs.
37 std::make_unsigned_t<INT> res =
38 CheckedNumeric<INT>(value).UnsignedAbs().ValueOrDie();
39
40 CHR* end = UNSAFE_TODO(outbuf + kOutputBufSize);
41 CHR* i = end;
42 do {
43 UNSAFE_TODO(--i);
44 DCHECK(i != outbuf);
45 *i = static_cast<CHR>((res % 10) + '0');
46 res /= 10;
47 } while (res != 0);
48 if (IsValueNegative(value)) {
49 UNSAFE_TODO(--i);
50 DCHECK(i != outbuf);
51 *i = static_cast<CHR>('-');
52 }
53 return STR(i, end);
54}

56// Utility to convert a character to a digit in a given base
57template <int BASE, typename CHAR>
58std::optional<uint8_t> CharToDigit(CHAR c) {
59 static_assert(1 <= BASE && BASE <= 36, "BASE needs to be in [1, 36]");
60 if (c >= '0' && c < '0' + std::min(BASE, 10)) {
61 return static_cast<uint8_t>(c - '0');
62 }
63
64 if (c >= 'a' && c < 'a' + BASE - 10) {
65 return static_cast<uint8_t>(c - 'a' + 10);
66 }
67
68 if (c >= 'A' && c < 'A' + BASE - 10) {
69 return static_cast<uint8_t>(c - 'A' + 10);
70 }
71
72 return std::nullopt;
73}

75template <typename Number, int kBase>
76class StringToNumberParser {
77 public:
78 struct Result {
79 Number value = 0;
80 bool valid = false;
81 };
82
83 static constexpr Number kMin = std::numeric_limits<Number>::min();
84 static constexpr Number kMax = std::numeric_limits<Number>::max();
85
86 // Sign provides:
87 // - a static function, CheckBounds, that determines whether the next digit
88 // causes an overflow/underflow
89 // - a static function, Increment, that appends the next digit appropriately
90 // according to the sign of the number being parsed.
91 template <typename Sign>
92 class Base {
93 public:
94 template <typename Iter>
95 static Result Invoke(Iter begin, Iter end) {
96 Number value = 0;
97
98 if (begin == end) {
99 return {value, false};
100 }
101
102 // Note: no performance difference was found when using template
103 // specialization to remove this check in bases other than 16
104 if (kBase == 16 && end - begin > 2 && *begin == '0' &&
105 (*(begin + 1) == 'x' || *(begin + 1) == 'X')) {
106 begin += 2;
107 }
108
109 for (Iter current = begin; current != end; ++current) {
110 std::optional<uint8_t> new_digit = CharToDigit<kBase>(*current);
111
112 if (!new_digit) {
113 return {value, false};
114 }
115
116 if (current != begin) {
117 Result result = Sign::CheckBounds(value, *new_digit);
118 if (!result.valid) {
119 return result;
120 }
121
122 value *= kBase;
123 }
124
125 value = Sign::Increment(value, *new_digit);
126 }
127 return {value, true};
128 }
129 };
130
131 class Positive : public Base<Positive> {
132 public:
133 static Result CheckBounds(Number value, uint8_t new_digit) {
134 if (value > static_cast<Number>(kMax / kBase) ||
135 (value == static_cast<Number>(kMax / kBase) &&
136 new_digit > kMax % kBase)) {
137 return {kMax, false};
138 }
139 return {value, true};
140 }
141 static Number Increment(Number lhs, uint8_t rhs) { return lhs + rhs; }
142 };
143
144 class Negative : public Base<Negative> {
145 public:
146 static Result CheckBounds(Number value, uint8_t new_digit) {
147 if (value < kMin / kBase ||
148 (value == kMin / kBase && new_digit > 0 - kMin % kBase)) {
149 return {kMin, false};
150 }
151 return {value, true};
152 }
153 static Number Increment(Number lhs, uint8_t rhs) { return lhs - rhs; }
154 };
155};

157template <typename Number, int kBase, typename CharT>
158auto StringToNumber(std::basic_string_view<CharT> input) {
159 using Parser = StringToNumberParser<Number, kBase>;
160 using Result = typename Parser::Result;
161
162 bool has_leading_whitespace = false;
163 auto begin = input.begin();
164 auto end = input.end();
165
166 while (begin != end && IsAsciiWhitespace(*begin)) {
167 has_leading_whitespace = true;
168 ++begin;
169 }
170
171 if (begin != end && *begin == '-') {
172 if (!std::numeric_limits<Number>::is_signed) {
173 return Result{0, false};
174 }
175
176 Result result = Parser::Negative::Invoke(begin + 1, end);
177 result.valid &= !has_leading_whitespace;
178 return result;
179 }
180
181 if (begin != end && *begin == '+') {
182 ++begin;
183 }
184
185 Result result = Parser::Positive::Invoke(begin, end);
186 result.valid &= !has_leading_whitespace;
187 return result;
188}

235template <typename STRING, typename CHAR>
236bool StringToDoubleImpl(STRING input, const CHAR* data, double& output) {
237 static double_conversion::StringToDoubleConverter converter(
238 double_conversion::StringToDoubleConverter::ALLOW_LEADING_SPACES |
239 double_conversion::StringToDoubleConverter::ALLOW_TRAILING_JUNK,
240 0.0, 0, nullptr, nullptr);
241
242 int processed_characters_count;
243 output = converter.StringToDouble(data, checked_cast<int>(input.size()),
244 &processed_characters_count);
245
246 // Cases to return false:
247 // - If the input string is empty, there was nothing to parse.
248 // - If the value saturated to HUGE_VAL.
249 // - If the entire string was not processed, there are either characters
250 // remaining in the string after a parsed number, or the string does not
251 // begin with a parseable number.
252 // - If the first character is a space, there was leading whitespace. Note
253 // that this checks using IsWhitespace(), which behaves differently for
254 // wide and narrow characters -- that is intentional and matches the
255 // behavior of the double_conversion library's whitespace-skipping
256 // algorithm.
257 return !input.empty() && output != HUGE_VAL && output != -HUGE_VAL &&
258 static_cast<size_t>(processed_characters_count) == input.size() &&
259 !IsWhitespace(input[0]);
260}

262template <typename Char, typename OutIter>
263static bool HexStringToByteContainer(std::string_view input, OutIter output) {
264 size_t count = input.size();
265 if (count == 0 || (count % 2) != 0) {
266 return false;
267 }
268 for (uintptr_t i = 0; i < count / 2; ++i) {
269 // most significant 4 bits
270 std::optional<uint8_t> msb = CharToDigit<16>(input[i * 2]);
271 // least significant 4 bits
272 std::optional<uint8_t> lsb = CharToDigit<16>(input[i * 2 + 1]);
273 if (!msb || !lsb) {
274 return false;
275 }
276 *(output++) = static_cast<Char>((*msb << 4) | *lsb);
277 }
278 return true;
279}
