381// unsafe_default_initialized
382// unsafe_default_initialized_cv
383//
384// An object which is explicitly convertible to any default-constructible type
385// and which, upon conversion, yields a default-initialized value of that type.
386//
387// https://en.cppreference.com/w/cpp/language/default_initialization
388//
389// For fundamental types, a default-initialized instance may have indeterminate
390// value. Reading an indeterminate value is undefined behavior but may offer a
391// performance optimization. When using an indeterminate value as a performance
392// optimization, it is best to be explicit.
393//
394// Useful as an escape hatch when enabling warnings or errors:
395// * gcc:
396// * uninitialized
397// * maybe-uninitialized
398// * clang:
399// * uninitialized
400// * conditional-uninitialized
401// * sometimes-uninitialized
402// * uninitialized-const-reference
403// * msvc:
404// * C4701: potentially uninitialized local variable used
405// * C4703: potentially uninitialized local pointer variable used
406//
407// Example:
408//
409// int local = folly::unsafe_default_initialized;
410// store_value_into_int_ptr(&value); // suppresses possible warning
411// use_value(value); // suppresses possible warning
412struct unsafe_default_initialized_cv {
413 KIWI_PUSH_WARNING
414 // MSVC requires warning disables to be outside of function definition
415 // Uninitialized local variable 'uninit' used
416 KIWI_MSVC_DISABLE_WARNING(4700)
417 // Potentially uninitialized local variable 'uninit' used
418 KIWI_MSVC_DISABLE_WARNING(4701)
419 // Potentially uninitialized local pointer variable 'uninit' used
420 KIWI_MSVC_DISABLE_WARNING(4703)
421 KIWI_GNU_DISABLE_WARNING("-Wuninitialized")
422 // Clang doesn't implement -Wmaybe-uninitialized and warns about it
423 KIWI_GCC_DISABLE_WARNING("-Wmaybe-uninitialized")
424 template <typename T>
425 KIWI_ERASE constexpr /* implicit */ operator T() const noexcept {
426#if defined(__cpp_lib_is_constant_evaluated)
427#if __cpp_lib_is_constant_evaluated >= 201811L
428#if !defined(__MSVC_RUNTIME_CHECKS)
429 if (!std::is_constant_evaluated()) {
430 T uninit;
431 return uninit;
432 }
433#endif // !defined(__MSVC_RUNTIME_CHECKS)
434#endif
435#endif
436 return T();
437 }
438 KIWI_POP_WARNING
439};
440inline constexpr unsafe_default_initialized_cv unsafe_default_initialized{};

442struct to_signed_fn {
443 template <typename..., typename T>
444 constexpr auto operator()(T const& t) const noexcept ->
445 typename std::make_signed<T>::type {
446 using S = typename std::make_signed<T>::type;
447 // note: static_cast<S>(t) would be more straightforward, but it would also
448 // be implementation-defined behavior and that is typically to be avoided;
449 // the following code optimized into the same thing, though
450 constexpr auto m = static_cast<T>(std::numeric_limits<S>::max());
451 return m < t ? -static_cast<S>(~t) + S{-1} : static_cast<S>(t);
452 }
453};

698// invocable_to
699// invocable_to_fn
700//
701// Given an invocable, returns an object which is implicitly convertible to the
702// type which the invocable returns when invoked with no arguments. Conversion
703// invokes the invocables and returns the value.
704//
705// The return object has unspecified type with the following semantics:
706// * It stores a decay-copy of the passed invocable.
707// * It defines four-way conversion operators. Each conversion operator purely
708// forwards to the invocable as forwarded-like the convertible, and has the
709// same exception specification and the same participation in overload
710// resolution as invocation of the invocable.
711//
712// Example:
713//
714// Given a setup:
715//
716// struct stable {
717// int value = 0;
718// stable() = default;
719// stable(stable const&); // expensive!
720// };
721// std::list<stable const> list;
722//
723// The goal is to insert a stable with a value of 7 to the back of the list.
724//
725// The obvious ways are expensive:
726//
727// stable obj;
728// obj.value = 7;
729// list.push_back(obj); // or variations with emplace_back or std::move
730//
731// With a lambda and copy elision optimization (NRVO), the expense remains:
732//
733// list.push_back(std::invoke([] {
734// stable obj;
735// obj.value = 7;
736// return obj;
737// }));
738//
739// But conversion, as done with this utility, makes this goal achievable.
740//
741// list.emplace_back(folly::invocable_to([] {
742// stable obj;
743// obj.value = 7;
744// return obj;
745// }));
746struct invocable_to_fn {
747 template <typename F, typename..., typename D = detail::decay_t<F>,
748 typename R = detail::invocable_to_convertible<D>,
749 std::enable_if_t<std::is_constructible<D, F&&>::value, int> = 0>
750 KIWI_ERASE constexpr R operator()(F&& f) const
751 noexcept(noexcept(R(static_cast<F&&>(f)))) {
752 return R(static_cast<F&&>(f));
753 }
754};
755inline constexpr invocable_to_fn invocable_to{};
